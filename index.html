R-01 Robots

Microbots

Robot controlado por Bluetooth con sensor ultrasónico (detección de obstáculos) basado en ESP32.

Club STEAM VCP • 21/07/2026
01 · LA IDEA

¿Qué problema resuelve o qué te dio ganas de hacerlo?

La idea de este proyecto es aprender sobre el control de actuadores, motores y la lectura de sensores ultrasónicos evitando obstáculos, utilizando el microcontrolador ESP32 y su Bluetooth integrado.

02 · FUNCIONAMIENTO

¿Cómo funciona?

El robot se conecta mediante Bluetooth al celular. Recibe comandos para moverse (adelante, atrás, izquierda, derecha) o regular su velocidad. Además, cuenta con un Modo Automático: al activarlo, el robot avanza solo y si el sensor ultrasónico detecta un obstáculo a menos de 25 cm, frena, retrocede y gira aleatoriamente para esquivarlo.

05 · INSTRUCTIVO

Cómo replicarlo, paso a paso

1
Conseguí los materiales y armá el chasis

Fijá los motores a la base. Revisá el pinout del ESP32 para verificar que los pines que vas a usar (16, 17, 18, 19, 22, 23) estén libres.

2
Hacé las conexiones

Conectá los 4 pines de control del puente H a los pines 16, 17, 18 y 19 del ESP32. Conectá el TRIG del ultrasónico al pin 22 y el ECHO al pin 23.

3
Cargá el programa en el ESP32

Copiá el código de la sección "Código" de abajo y subilo a la placa. Recordá tener instaladas las tarjetas ESP32 en tu IDE de Arduino.

4
Controlalo desde el celular

Conectate por Bluetooth al dispositivo llamado "Microbots".Usa la app que diseñe para poder manejarlo.Para actiar el modo automatico tienes que tocar el boton que dice apagar

06 · CÓDIGO Y ARCHIVOS

¿Dónde está el código o los archivos del proyecto?

Código fuente principal (ESP32):

#include "BluetoothSerial.h"

#include "BluetoothSerial.h"
BluetoothSerial SerialBT;

const int MA_F = 16;
const int MA_B = 17;
const int MB_F = 18;
const int MB_B = 19;

const int TRIG_PIN = 22;
const int ECHO_PIN = 23; 

const int freq       = 1000;
const int resolution = 8;

int velocity        = 180; 
const int autoSpeed = 180;

char lastCmd         = 'S';
bool autoMode        = false;
String velBuffer     = "";
bool   readingVel    = false;

int autoState = 0; 
unsigned long autoTimer = 0;
unsigned long lastSonarTime = 0;
char randomTurnDir = 'R'; 

void movement(char c);
void stopAll();
bool isValidCmd(char c);
int  getDistance();
void processBluetooth();
void processAutoMode();

void setup() {
  Serial.begin(115200);
  SerialBT.begin("Microbots");

  randomSeed(analogRead(34)); 

  ledcAttach(MA_F, freq, resolution);
  ledcAttach(MA_B, freq, resolution);
  ledcAttach(MB_F, freq, resolution);
  ledcAttach(MB_B, freq, resolution);

  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);

  movement('S');
  Serial.println("Sistema iniciado.");
}

void loop() {
  processBluetooth();

  if (autoMode) {
    processAutoMode();
  }
}

void processBluetooth() {
  while (SerialBT.available()) {
    char cmd = SerialBT.read();

    if (cmd == '\r' || cmd == '\n') {
      if (readingVel && velBuffer.length() > 0) {
        readingVel = false;
        velocity  = constrain(velBuffer.toInt(), 0, 255);
        velBuffer = "";
        if (!autoMode && lastCmd != 'S' && isValidCmd(lastCmd)) movement(lastCmd);
      }
      continue;
    }

    cmd = toupper(cmd);

    if (readingVel) {
      if (isDigit(cmd)) {
        velBuffer += cmd;
        if (velBuffer.length() >= 3) {
          readingVel = false;
          velocity   = constrain(velBuffer.toInt(), 0, 255);
          velBuffer  = "";
          if (!autoMode && lastCmd != 'S' && isValidCmd(lastCmd)) movement(lastCmd);
        }
      } else {
        readingVel = false;
        velBuffer  = "";
      }
      continue;
    }

    if (cmd == 'V') {
      readingVel = true;
      velBuffer  = "";
      continue;
    }

    if (!isValidCmd(cmd)) continue;

    if (cmd == 'O') {
      autoMode = !autoMode; 
      if (autoMode) {
        autoState = 0; 
        lastCmd = 'O';
      } else {
        movement('S');
        lastCmd = 'S';
      }
      continue;
    } 

    else {
      if (autoMode) {
        autoMode = false; 
      }
      lastCmd = cmd;
      movement(cmd);
    }
  }
}

void processAutoMode() {
  unsigned long currentMillis = millis();

  switch (autoState) {
    case 0: // AVANZAR Y MEDIR
      if (currentMillis - lastSonarTime > 60) {
        lastSonarTime = currentMillis;
        int distancia = getDistance();
        
        // Detección a 25 cm
        if (distancia > 0 && distancia <= 25) {
          movement('S');
          randomTurnDir = (random(0, 2) == 0) ? 'L' : 'R';
          autoState = 1;
          autoTimer = currentMillis;
        } else {
          if (lastCmd != 'F') {
            movement('F');
            lastCmd = 'F';
          }
        }
      }
      break;

    case 1: // FRENO SECO
      if (currentMillis - autoTimer >= 150) {
        movement('B'); 
        autoState = 2;
        autoTimer = currentMillis;
      }
      break;

    case 2: // RETROCEDER
      if (currentMillis - autoTimer >= 400) {
        movement(randomTurnDir); 
        autoState = 3;
        autoTimer = currentMillis;
      }
      break;

    case 3: // GIRAR
      if (currentMillis - autoTimer >= 450) {
        movement('S'); 
        autoState = 4;
        autoTimer = currentMillis;
      }
      break;

    case 4: // ESTABILIZAR Y REANUDAR
      if (currentMillis - autoTimer >= 100) {
        autoState = 0; 
        lastCmd = 'S'; 
      }
      break;
  }
}

bool isValidCmd(char c) {
  return (c == 'F' || c == 'B' || c == 'L' || c == 'R' || c == 'S' || c == 'O');
}

int getDistance() {
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);
  
  long duration = pulseIn(ECHO_PIN, HIGH, 30000); 
  
  if (duration == 0) {
    // Reseteo rápido del pin si el sensor se cuelga
    pinMode(ECHO_PIN, OUTPUT);
    digitalWrite(ECHO_PIN, LOW);
    delayMicroseconds(100);
    pinMode(ECHO_PIN, INPUT);
    return 999; 
  }
  
  return duration * 0.034 / 2;
}

// ============================================================
//  CONTROL DE MOTORES
// ============================================================
void stopAll() {
  ledcWrite(MA_F, 0);
  ledcWrite(MB_F, 0);
  ledcWrite(MA_B, 0);
  ledcWrite(MB_B, 0);
}

void movement(char c) {
  stopAll();
  
  int activeSpeed = autoMode ? autoSpeed : velocity;

  switch (c) {
    case 'F': // Adelante
      ledcWrite(MA_F, activeSpeed);
      ledcWrite(MB_F, activeSpeed);
      break;
    case 'B': // Atrás
      ledcWrite(MA_B, activeSpeed);
      ledcWrite(MB_B, activeSpeed);
      break;
    case 'L': // Izquierda (Motor Dcho Adelante, Motor Izq Atrás)
      ledcWrite(MA_B, activeSpeed);
      ledcWrite(MB_F, activeSpeed);
      break;
    case 'R': // Derecha (Motor Izq Adelante, Motor Dcho Atrás)
      ledcWrite(MA_F, activeSpeed);
      ledcWrite(MB_B, activeSpeed);
      break;
    case 'S': // Detener
      // stopAll() ya se ejecutó arriba
      break;
  }
}
07 · APRENDIZAJES

Desafíos y qué aprendiste

Una de las mejoras claves fue crear una "máquina de estados" usando millis() en lugar de delay(). Esto permite que el robot esquive obstáculos fluidamente sin que el código se quede bloqueado, pudiendo recibir comandos Bluetooth en todo momento.
